two-sum

Time Complexity :O(N)

Space Complexity:O(1)

func twoSum(nums []int, target int) []int { visited:=make(map[int]int) ret:= make([]int,2); var t int for i,x :=range(nums){ t=target-x val,ok:=visited[t]; if ok{ ret[0]=i ret[1]=val return ret } visited[x]=i } return ret; }

func isPalindrome(x int) bool { if(x<0){ return false } t:=strconv.Itoa(x) for i :=range(len(t)){ if t[i]!=t[len(t)-i-1]{ return false; } } return true }
